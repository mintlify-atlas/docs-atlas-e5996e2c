## Overview

This page provides practical examples for interacting with the Mango Damage Detection API in various scenarios and programming languages.

## Damage Types

The model can detect the following mango conditions:
- **Anthracnose**: Fungal disease causing dark lesions
- **Bacterial-Black-spot**: Bacterial infection with black spots
- **Fruitfly**: Damage from fruit fly infestation
- **Healthy-mango**: No visible damage
- **Others**: Unclassified damage or conditions

## Basic API Request

## Python Examples

### Single Image Prediction

```python single_prediction.py
import requests

def predict_mango_damage(image_path, api_url="http://localhost:8001/predict/"):
    """
    Send an image to the Mango Damage Detection API.
    
    Args:
        image_path: Path to the mango image file
        api_url: FastAPI endpoint URL
        
    Returns:
        dict: Prediction result
    """
    try:
        with open(image_path, 'rb') as image_file:
            files = {'file': (image_path, image_file, 'image/jpeg')}
            response = requests.post(api_url, files=files)
            
        if response.status_code == 200:
            result = response.json()
            print(f"Prediction: {result['prediction']}")
            return result
        else:
            print(f"Error: {response.status_code} - {response.text}")
            return None
            
    except FileNotFoundError:
        print(f"Error: Image file '{image_path}' not found")
        return None
    except requests.exceptions.RequestException as e:
        print(f"Request failed: {e}")
        return None

# Usage
result = predict_mango_damage('mango_anthracnose.jpg')
```

### Batch Processing Multiple Images

```python batch_processing.py
import requests
import os
from pathlib import Path
import time

def batch_predict(image_directory, api_url="http://localhost:8001/predict/"):
    """
    Process multiple mango images in a directory.
    
    Args:
        image_directory: Path to directory containing images
        api_url: FastAPI endpoint URL
        
    Returns:
        list: Results for all images
    """
    results = []
    image_extensions = {'.jpg', '.jpeg', '.png', '.bmp'}
    
    # Get all image files in directory
    image_files = [
        f for f in Path(image_directory).iterdir()
        if f.suffix.lower() in image_extensions
    ]
    
    print(f"Processing {len(image_files)} images...")
    
    for idx, image_path in enumerate(image_files, 1):
        try:
            with open(image_path, 'rb') as image_file:
                files = {'file': (image_path.name, image_file, 'image/jpeg')}
                response = requests.post(api_url, files=files)
                
            if response.status_code == 200:
                prediction = response.json()['prediction']
                result = {
                    'filename': image_path.name,
                    'prediction': prediction,
                    'status': 'success'
                }
                print(f"[{idx}/{len(image_files)}] {image_path.name}: {prediction}")
            else:
                result = {
                    'filename': image_path.name,
                    'error': f"{response.status_code} - {response.text}",
                    'status': 'error'
                }
                print(f"[{idx}/{len(image_files)}] {image_path.name}: Error {response.status_code}")
                
            results.append(result)
            
            # Small delay to avoid overwhelming the server
            time.sleep(0.1)
            
        except Exception as e:
            print(f"[{idx}/{len(image_files)}] {image_path.name}: Exception - {e}")
            results.append({
                'filename': image_path.name,
                'error': str(e),
                'status': 'error'
            })
    
    return results

# Usage
results = batch_predict('./mango_images')

# Print summary
successful = sum(1 for r in results if r['status'] == 'success')
print(f"\nProcessed: {len(results)} images")
print(f"Successful: {successful}")
print(f"Failed: {len(results) - successful}")

# Count predictions by type
from collections import Counter
predictions = [r['prediction'] for r in results if r['status'] == 'success']
counts = Counter(predictions)
print("\nPrediction Summary:")
for damage_type, count in counts.items():
    print(f"  {damage_type}: {count}")
```

### With Advanced Error Handling

```python robust_client.py
import requests
from requests.adapters import HTTPAdapter
from urllib3.util.retry import Retry
import logging

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class MangoDamageDetector:
    """Robust client for Mango Damage Detection API."""
    
    def __init__(self, api_url="http://localhost:8001", timeout=30, max_retries=3):
        self.api_url = api_url
        self.timeout = timeout
        self.session = self._create_session(max_retries)
        
    def _create_session(self, max_retries):
        """Create a session with retry logic."""
        session = requests.Session()
        retry = Retry(
            total=max_retries,
            backoff_factor=0.5,
            status_forcelist=[500, 502, 503, 504]
        )
        adapter = HTTPAdapter(max_retries=retry)
        session.mount('http://', adapter)
        session.mount('https://', adapter)
        return session
    
    def predict(self, image_path):
        """Predict damage type for a mango image."""
        try:
            with open(image_path, 'rb') as image_file:
                files = {'file': (image_path, image_file, 'image/jpeg')}
                response = self.session.post(
                    f"{self.api_url}/predict/",
                    files=files,
                    timeout=self.timeout
                )
                
            response.raise_for_status()
            result = response.json()
            logger.info(f"Prediction for {image_path}: {result['prediction']}")
            return result
            
        except requests.exceptions.Timeout:
            logger.error(f"Request timeout after {self.timeout} seconds")
            raise
        except requests.exceptions.HTTPError as e:
            logger.error(f"HTTP error: {e}")
            raise
        except requests.exceptions.RequestException as e:
            logger.error(f"Request failed: {e}")
            raise
        except FileNotFoundError:
            logger.error(f"Image file not found: {image_path}")
            raise
        except KeyError:
            logger.error("Invalid response format from API")
            raise
    
    def health_check(self):
        """Check if the API is accessible."""
        try:
            response = self.session.get(f"{self.api_url}/docs", timeout=5)
            return response.status_code == 200
        except:
            return False

# Usage
detector = MangoDamageDetector(api_url="http://localhost:8001")

if detector.health_check():
    print("API is accessible")
    result = detector.predict('healthy_mango.jpg')
    print(f"Result: {result}")
else:
    print("API is not accessible")
```

## Integration Examples

### Flask Application Integration

```python flask_integration.py
from flask import Flask, request, render_template, jsonify
import requests

app = Flask(__name__)
FAST_API_URL = "http://localhost:8001/predict/"

@app.route('/')
def index():
    return render_template('upload.html')

@app.route('/predict', methods=['POST'])
def predict():
    if 'file' not in request.files:
        return jsonify({'error': 'No file uploaded'}), 400
    
    file = request.files['file']
    
    if file.filename == '':
        return jsonify({'error': 'Empty filename'}), 400
    
    try:
        # Forward file to FastAPI
        files = {'file': (file.filename, file.read(), file.content_type)}
        response = requests.post(FAST_API_URL, files=files, timeout=30)
        
        if response.status_code == 200:
            prediction = response.json()['prediction']
            return jsonify({
                'success': True,
                'prediction': prediction,
                'filename': file.filename
            })
        else:
            return jsonify({
                'success': False,
                'error': f"API error: {response.status_code}"
            }), response.status_code
            
    except requests.exceptions.Timeout:
        return jsonify({'error': 'Request timeout'}), 504
    except Exception as e:
        return jsonify({'error': str(e)}), 500

if __name__ == '__main__':
    app.run(debug=True, port=5000)
```

### Standalone FastAPI Client

```python fastapi_client.py
from fastapi import FastAPI, File, UploadFile, HTTPException
from fastapi.responses import JSONResponse
import requests
import httpx

app = FastAPI()
ML_API_URL = "http://localhost:8001/predict/"

@app.post("/analyze")
async def analyze_mango(file: UploadFile = File(...)):
    """
    Proxy endpoint that adds additional business logic.
    """
    try:
        # Read file contents
        contents = await file.read()
        
        # Validate file size (e.g., max 5MB)
        if len(contents) > 5 * 1024 * 1024:
            raise HTTPException(status_code=413, detail="File too large")
        
        # Forward to ML API using async client
        async with httpx.AsyncClient() as client:
            files = {'file': (file.filename, contents, file.content_type)}
            response = await client.post(ML_API_URL, files=files, timeout=30.0)
            
        if response.status_code == 200:
            prediction = response.json()['prediction']
            
            # Add business logic
            severity = classify_severity(prediction)
            
            return JSONResponse(content={
                'prediction': prediction,
                'severity': severity,
                'filename': file.filename,
                'recommendations': get_recommendations(prediction)
            })
        else:
            raise HTTPException(
                status_code=response.status_code,
                detail="ML API request failed"
            )
            
    except httpx.TimeoutException:
        raise HTTPException(status_code=504, detail="ML API timeout")
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

def classify_severity(prediction):
    """Classify damage severity."""
    severity_map = {
        'Healthy-mango': 'none',
        'Others': 'low',
        'Fruitfly': 'medium',
        'Bacterial-Black-spot': 'high',
        'Anthracnose': 'high'
    }
    return severity_map.get(prediction, 'unknown')

def get_recommendations(prediction):
    """Get treatment recommendations."""
    recommendations = {
        'Anthracnose': 'Apply fungicide treatment and improve ventilation',
        'Bacterial-Black-spot': 'Remove infected fruits and apply copper-based bactericide',
        'Fruitfly': 'Use fruit fly traps and maintain orchard hygiene',
        'Healthy-mango': 'Continue regular monitoring and maintenance',
        'Others': 'Consult with agricultural expert for proper diagnosis'
    }
    return recommendations.get(prediction, 'No specific recommendation')
```

## Error Handling Examples

### Handling Common Errors

```python error_handling.py
import requests
from requests.exceptions import RequestException, Timeout, ConnectionError

def predict_with_error_handling(image_path, api_url="http://localhost:8001/predict/"):
    """
    Comprehensive error handling example.
    """
    try:
        with open(image_path, 'rb') as image_file:
            files = {'file': (image_path, image_file, 'image/jpeg')}
            response = requests.post(api_url, files=files, timeout=30)
            
        # Check for HTTP errors
        if response.status_code == 200:
            return {'success': True, 'data': response.json()}
        elif response.status_code == 422:
            return {
                'success': False,
                'error': 'Invalid file format or corrupted image'
            }
        elif response.status_code == 500:
            return {
                'success': False,
                'error': 'Server error - model prediction failed'
            }
        else:
            return {
                'success': False,
                'error': f"HTTP {response.status_code}: {response.text}"
            }
            
    except FileNotFoundError:
        return {
            'success': False,
            'error': f"Image file not found: {image_path}"
        }
    except Timeout:
        return {
            'success': False,
            'error': 'Request timeout - server may be overloaded'
        }
    except ConnectionError:
        return {
            'success': False,
            'error': 'Cannot connect to API - check if server is running'
        }
    except RequestException as e:
        return {
            'success': False,
            'error': f"Request failed: {str(e)}"
        }
    except Exception as e:
        return {
            'success': False,
            'error': f"Unexpected error: {str(e)}"
        }

# Usage
result = predict_with_error_handling('mango_sample.jpg')

if result['success']:
    print(f"Prediction: {result['data']['prediction']}")
else:
    print(f"Error: {result['error']}")
```

## Performance Considerations

<Tip>
  For production use:
  - Implement connection pooling for batch processing
  - Add request timeouts (30 seconds recommended)
  - Use async/await for concurrent requests
  - Implement circuit breaker pattern for resilience
  - Cache results for identical images (optional)
</Tip>

<Warning>
  The FastAPI service processes images synchronously. For high-volume applications, consider:
  - Load balancing across multiple FastAPI instances
  - Implementing a task queue (Celery, RQ)
  - Using GPU acceleration for faster predictions
</Warning>
